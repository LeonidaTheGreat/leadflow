## 🔧 Report Generation Script

```python
#!/usr/bin/env python3
"""
Weekly Report Generator for AI Lead Response System
Run every Monday at 9 AM EST via cron or scheduled task
"""

import os
import json
from datetime import datetime, timedelta
from posthog import Posthog
import requests

# Configuration
POSTHOG_API_KEY = os.environ['POSTHOG_API_KEY']
POSTHOG_PROJECT_ID = os.environ['POSTHOG_PROJECT_ID']
SUPABASE_URL = os.environ['SUPABASE_URL']
SUPABASE_KEY = os.environ['SUPABASE_KEY']

def get_date_ranges():
    """Get current and previous week date ranges"""
    today = datetime.now()
    
    # This week (Monday-Sunday)
    this_week_end = today - timedelta(days=today.weekday() + 1)
    this_week_start = this_week_end - timedelta(days=6)
    
    # Last week
    last_week_end = this_week_start - timedelta(days=1)
    last_week_start = last_week_end - timedelta(days=6)
    
    return {
        'this_week': (this_week_start, this_week_end),
        'last_week': (last_week_start, last_week_end)
    }

def generate_report():
    """Generate the full weekly report"""
    dates = get_date_ranges()
    
    # Build report structure
    report = {
        'week_of': dates['this_week'][0].strftime('%Y-%m-%d'),
        'generated_at': datetime.now().isoformat(),
        'product_metrics': {
            'leads_received': 0,
            'response_time_avg': 0,
            'delivery_rate': 0,
            'booking_conversion': 0
        },
        'business_metrics': {
            'mrr': 0,
            'new_customers': 0,
            'churned': 0
        },
        'alerts': {
            'critical': [],
            'warnings': []
        }
    }
    
    return report

def main():
    """Main entry point"""
    report = generate_report()
    
    # Save to file
    filename = f"weekly_report_{report['week_of']}.json"
    with open(filename, 'w') as f:
        json.dump(report, f, indent=2)
    
    print(f"✅ Report generated: {filename}")

if __name__ == '__main__':
    main()
```

## 🔄 Automation Setup

### Option 1: GitHub Actions

```yaml
# .github/workflows/weekly-report.yml
name: Generate Weekly Report

on:
  schedule:
    # Every Monday at 9 AM EST (14:00 UTC)
    - cron: '0 14 * * 1'
  workflow_dispatch:  # Manual trigger

jobs:
  generate-report:
    runs-on: ubuntu-latest
    
    steps:
      - uses: actions/checkout@v3
      
      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.11'
      
      - name: Install dependencies
        run: |
          pip install posthog requests
      
      - name: Generate report
        env:
          POSTHOG_API_KEY: ${{ secrets.POSTHOG_API_KEY }}
          POSTHOG_PROJECT_ID: ${{ secrets.POSTHOG_PROJECT_ID }}
          SUPABASE_URL: ${{ secrets.SUPABASE_URL }}
          SUPABASE_KEY: ${{ secrets.SUPABASE_KEY }}
          SENDGRID_API_KEY: ${{ secrets.SENDGRID_API_KEY }}
          SLACK_WEBHOOK_URL: ${{ secrets.SLACK_WEBHOOK_URL }}
        run: |
          python scripts/generate_weekly_report.py
      
      - name: Upload report artifact
        uses: actions/upload-artifact@v3
        with:
          name: weekly-report
          path: weekly_report_*.json
```
